## 选择排序

- 选择排序，不需要像冒泡排序一样每次都换位置

```
let arr = [33, 21, 9, 23, 14, 55, 75, 4, 63, 12, 77];

function selectSort(arr){
  let min;
  for(let i=0;i<arr.length;i++){
    min = i;
    for(let j=i+1;j<arr.length;j++){
      if(arr[min]>arr[j]){
        min = j;
      }
    }
    if(min!=i){
      [arr[i],arr[min]]=[arr[min],arr[i]];
    }
  }
  return arr;
}

console.log(selectSort(arr));
```
